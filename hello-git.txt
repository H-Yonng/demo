$ mkdir name	    	    //创建文件夹
$ cd file			          //打开文件夹
$ cd ..			            //退回目录上一层
$ pwd 			            //查看当前目录
$ git init 			        //本地仓库
$ git touch file		    //创建文件
$ notepad		            //打开记事本
$ git add file		      //添加到暂存区	
$ git status 		        //查看仓库当前的状态
$ git commit -m "注释" 	//提交
$ git diff file		      //查看文件修改内容
$ git reset HEAD file  	//git reset命令既可以回退版本，
$ git reset	 file		    也可以把暂存区的修改回退到工作区
$ git checkout -- file 	////把文件在工作区的修改全部撤销
$ git diff HEAD -- file	//查看新旧版区别，HEAD表示最新版本
$ git log file		      //查看日志
$ rm file			          //删除文件
$ git rm -- file 		    //删除版本库中的文件
$ git checkout -- file	//用版本库里的版本替换工作区的版本
$ git remote add origin git@github.com
$ git push -u origin master	
$ git push origin mater	//推送到远程库
